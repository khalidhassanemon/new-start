#include<bits/stdc++.h>
#define M 100
bool marked[M];
using namespace std;
void sieve(int n1)
{
    for(int i=3; i*i<=n1; i+=2)
    {
        if(marked[i]==false)
        {
            for(int j=i*i; j<=n1; j+=i+i)
            {
                marked[j]=true;
            }
        }
    }
}
bool isPrime(int n)
{
    if (n < 2)
        return false;
    if (n == 2)
        return true;
    if (n % 2 == 0)
        return false;
    if(marked[n]==false)
        return true;
    else
        return false;

}
int main()
{
    int n;
    while(cin>>n)
    {
        if(n==0)
            break;
        vector<int>v1,v2;
        sieve(n);
        for(int i=1; i<=n; i++)
        {
            if(isPrime(i)==true)
                v1.push_back(i);
        }
        int x=0,y=0,z=0,count=0;
        for(int i=0; i<v1.size(); i++)
        {
            x=v1[i];
            z=n;
            if((n/x)>0)
            {
                while(z>0)
                {
                    z/=x;
                    y+=z;
                }
                v2.push_back(y);
                x=0,y=0;
            }
            else
                break;
        }
        printf("%3d! =", n);
        for(int i=0; i<v2.size(); i++)
        {
            count++;
            if(count>15)
            {
                printf("\n      ");
                count=0;
            }
            printf("%3d",v2[i]);
        }
        cout<<""<<endl;

    }
    return 0;
}
